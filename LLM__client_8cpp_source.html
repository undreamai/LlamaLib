130X509_STORE *load_client_cert(const std::string &cert_str)
131{
132 BIO *mem = BIO_new_mem_buf(cert_str.data(), (int)cert_str.size());
133 if (!mem)
134 {
135 return nullptr;
136 }
137
138 auto inf = PEM_X509_INFO_read_bio(mem, nullptr, nullptr, nullptr);
139 if (!inf)
140 {
141 return nullptr;
142 }
143
144 auto cts = X509_STORE_new();
145 if (cts)
146 {
147 for (auto i = 0; i < static_cast<int>(sk_X509_INFO_num(inf)); i++)
148 {
149 auto itmp = sk_X509_INFO_value(inf, i);
150 if (!itmp)
151 {
152 continue;
153 }
154
155 if (itmp->x509)
156 {
157 X509_STORE_add_cert(cts, itmp->x509);
158 }
159 if (itmp->crl)
160 {
161 X509_STORE_add_crl(cts, itmp->crl);
162 }
163 }
164 }
165
166 sk_X509_INFO_pop_free(inf, X509_INFO_free);
167 BIO_free(mem);
168 return cts;
169}

171std::string LLMClient::post_request(
172 const std::string &path,
173 const json &payload,
174 CharArrayFn callback,
175 bool callbackWithJSON)
176{
177 StreamingContext context;
178 context.callback = callback;
179
180 json body = payload;
181 bool stream = callback != nullptr;
182 if (body.contains("stream"))
183 stream = body["stream"];
184 else
185 body["stream"] = stream;
186
187 httplib::Headers headers = {
188 {"Content-Type", "application/json"},
189 {"Accept", stream ? "text/event-stream" : "application/json"},
190 {"Cache-Control", "no-cache"}};
191
192 httplib::Request req;
193 req.method = "POST";
194 req.path = "/" + path;
195 req.headers = headers;
196 req.body = body.dump();
197
198 std::string concat_string = "";
199 std::vector<int> concat_tokens;
200 req.content_receiver = [&](const char *data, size_t data_length, uint64_t /*offset*/, uint64_t /*total_length*/)
201 {
202 if (stream)
203 {
204 std::string chunk_str(data, data_length);
205 // remove preceding "data: "
206 const std::string prefix = "data: ";
207 if (chunk_str.rfind(prefix, 0) == 0)
208 {
209 chunk_str.erase(0, prefix.length());
210 }
211 // Remove any trailing newlines or carriage returns
212 while (!chunk_str.empty() && (chunk_str.back() == '\n' || chunk_str.back() == '\r'))
213 {
214 chunk_str.pop_back();
215 }
216
217 json data_json = json::parse(chunk_str);
218 if (context.callback != nullptr)
219 {
220 if (callbackWithJSON)
221 {
222 context.callback(chunk_str.c_str());
223 }
224 else
225 {
226 if (data_json.contains("content"))
227 context.callback(data_json["content"].get<std::string>().c_str());
228 }
229 }
230
231 if (data_json.contains("content"))
232 {
233 concat_string += data_json["content"].get<std::string>();
234 }
235 if (data_json.contains("tokens"))
236 {
237 for (const auto &tok : data_json["tokens"])
238 {
239 concat_tokens.push_back(tok.get<int>());
240 }
241 }
242
243 json concat_data = data_json;
244 concat_data["content"] = concat_string;
245 concat_data["tokens"] = concat_tokens;
246 context.buffer = concat_data.dump();
247 }
248 else
249 {
250 std::string chunk_str(data, data_length);
251 context.buffer += chunk_str;
252 }
253 return true;
254 };
255
256 bool https = false;
257 std::string host;
258 if (url.rfind("https://", 0) == 0)
259 {
260 host = url.substr(8);
261 https = true;
262 }
263 else
264 {
265 host = url.rfind("http://", 0) == 0 ? url.substr(7) : url;
266 }
267
268 bool ok = false;
269 if (https && SSL_cert != "")
270 {
271 httplib::SSLClient cli(host.c_str(), port);
272 cli.set_ca_cert_store(load_client_cert(SSL_cert));
273 ok = cli.send(req);
274 }
275 else
276 {
277 httplib::Client cli(host.c_str(), port);
278 ok = cli.send(req);
279 }
280
281 if (!ok)
282 {
283 std::string error = "HTTP POST streaming request failed";
284 std::cerr << error << std::endl;
285 context.buffer = error;
286 }
287
288 return context.buffer;
289}
